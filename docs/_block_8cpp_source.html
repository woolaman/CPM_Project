133void Block::Segment1()
134{
135 // 自动寻峰, 奇异值分解, 寻找规则峰位作为平均值移动算法的迭代初始位置
136 // 平均值移动算法，寻找peaks
137
138 qDebug() << "SVD start.";
139 gsl_matrix* A = gsl_matrix_alloc(nPixel, nPixel);
140 gsl_matrix* V = gsl_matrix_alloc(nPixel, nPixel);
141 gsl_vector* S = gsl_vector_alloc(nPixel);
142 gsl_vector* workspace = gsl_vector_alloc(nPixel);
143
144 for (int i = 0; i < nPixel; ++i)
145 {
146 for (int j = 0; j < nPixel; ++j)
147 {
148 gsl_matrix_set(A, i, j, m_I0(i, j));
149 }
150 }
151
152 gsl_linalg_SV_decomp(A, V, S, workspace);
153
154 cv::Mat_<qreal> UMat = cv::Mat::zeros(nPixel, nPixel, CV_64FC1);
155 cv::Mat_<qreal> SMat = cv::Mat::zeros(nPixel, nPixel, CV_64FC1);
156 cv::Mat_<qreal> VMat = cv::Mat::zeros(nPixel, nPixel, CV_64FC1);
157
158 for (int i = 0; i < nPixel; ++i)
159 {
160 for (int j = 0; j < nPixel; ++j)
161 {
162 UMat(i, j) = gsl_matrix_get(A, i, j);
163 VMat(i, j) = gsl_matrix_get(V, i, j);
164 }
165 }
166
167 SMat(0, 0) = gsl_vector_get(S, 0);
168
169 cv::Mat I1 = UMat*SMat*VMat.t();
170
171 gsl_matrix_free(A);
172 gsl_matrix_free(V);
173 gsl_vector_free(S);
174 gsl_vector_free(workspace);
175 qDebug() << "SVD done. ";
176
177 cv::Mat col_sum, row_sum;
178 cv::reduce(I1, col_sum, 0, cv::REDUCE_SUM);
179 cv::reduce(I1, row_sum, 1, cv::REDUCE_SUM);
180
181 QVector<qreal> colSumVec = {col_sum.begin<qreal>(), col_sum.end<qreal>()};
182 QVector<qreal> rowSumVec = {row_sum.begin<qreal>(), row_sum.end<qreal>()};
183 QVector<int> peak_x_index = FindPeaks(colSumVec, nCrystal);
184 QVector<int> peak_y_index = FindPeaks(rowSumVec, nCrystal);
185
186 for (int i = 0; i < nCrystal; ++i)
187 {
188 for (int j= 0; j < nCrystal; ++j)
189 {
190 m_pt(i, j)[0] = peak_x_index[j];
191 m_pt(i, j)[1] = peak_y_index[i];
192 }
193 }
194 qDebug() << QString::fromLocal8Bit("所有晶体初始位置设置完毕。");
195
196 // step1，每组整体做平均值移动
197 for (int iRow = 0; iRow < m_num1; ++iRow)
198 {
199 for (int iCol = 0; iCol < m_num1; ++iCol)
200 {
201 // 求 4 个点的重心的初始位置
202 cv::Point2f c0(0, 0);
203 for (int i = 0; i < m_num2; ++i)
204 {
205 for (int j = 0; j < m_num2; ++j)
206 {
207 c0 += cv::Point2f(m_pt(iRow*m_num2+i, iCol*m_num2+j)[0],
208 m_pt(iRow*m_num2+i, iCol*m_num2+j)[1]);
209 }
210 }
211
212 cv::Point2f c = c0/(m_num2*m_num2); // 初始位置
213 cv::Point2f pre_c(0, 0);
214
215 while (cv::norm(pre_c-c) > 1) // 迭代移动重心
216 {
217 pre_c = c;
218 c = cv::Point2f(0, 0);
219
220 // 计算重心
221 for (int i = 0; i < m_num2; ++i)
222 {
223 for (int j = 0; j < m_num2; ++j)
224 {
225 cv::Point2f x(m_pt(iRow*m_num2+i, iCol*m_num2+j)[0],
226 m_pt(iRow*m_num2+i, iCol*m_num2+j)[1]);
227
228 // 累加项 // 附近的点进行累加
229 cv::Point2f u(0, 0);
230 qreal d = 0;
231 int r = 6; // 累加范围
232 for (int n = x.y-r; n <= x.y+r; ++n)
233 {
234 for (int m = x.x-r; m <= x.x+r; ++m)
235 {
236 if (cv::norm(x-cv::Point2f(m, n)) <= r)
237 {
238 u += m_I0(n, m)*cv::Point2f(m, n);
239 d += m_I0(n, m);
240 }
241 }
242 }
243
244 if(d==0)
245 {
246 qDebug() << "迭代过程中，分母 d = 0 !!! Bug !!!";
247 }
248 else
249 {
250 c += u/d;
251 }
252 }
253 }
254
255 c = c/(m_num2*m_num2);
256
257 for (int i = 0; i < m_num2; ++i)
258 {
259 for (int j = 0; j < m_num2; ++j)
260 {
261 m_pt(iRow*m_num2+i, iCol*m_num2+j)[0] += qRound(c.x - pre_c.x);
262 m_pt(iRow*m_num2+i, iCol*m_num2+j)[1] += qRound(c.y - pre_c.y);
263 }
264 }
265 }
266 }
267 }
268 qDebug() << QString::fromLocal8Bit("第一次平均值移动迭代，每组峰位移动完成。");
269
270 // step2, all peak 做平均值移动, 使得每个峰做小范围移动
271 for (int iRow = 0; iRow < nCrystal; ++iRow)
272 {
273 for (int iCol = 0; iCol < nCrystal; ++iCol)
274 {
275 cv::Point2f x(m_pt(iRow, iCol)[0], m_pt(iRow, iCol)[1]);
276 cv::Point2f pre_x(0, 0);
277 while(cv::norm(pre_x-x)>0.1)
278 {
279 pre_x = x;
280 cv::Point2f u(0, 0);
281 qreal d = 0;
282
283 int r = 3;
284 for (int n = x.y-r; n <= x.y+r; ++n)
285 {
286 for (int m = x.x-r; m <= x.x+r; ++m)
287 {
288 if (cv::norm(x-cv::Point2f(m, n)) <= r)
289 {
290 //qreal k = cv::norm(x-cv::Point2f(m, n))/pow((2*r+1), 2);
291 //qreal g = (1/sqrt(2*M_PI))*exp(-0.5*k);
292 qreal g = 1;
293 u += m_I0(n, m)*g*cv::Point2f(m, n);
294 d += m_I0(n, m)*g;
295 }
296 }
297 }
298
299 if (d==0)
300 {
301 qDebug() << "迭代过程中，分母 d = 0 !!! Bug !!!";
302 }
303 else
304 {
305 x = u/d;
306 }
307 }
308 m_pt(iRow, iCol)[0] = qRound(x.x);
309 m_pt(iRow, iCol)[1] = qRound(x.y);
310 }
311 }
312 qDebug() << QString::fromLocal8Bit("第二次平均值移动迭代。所有峰位寻找完成。");
313}

316void Block::Segment2() // maximum method to find peaks
317{
318 // 使用 OpenCV 的平滑函数对数据进行平滑处理
319 cv::Mat_<qreal> I1;
320 cv::blur(m_I0, I1, cv::Size(5, 5));
321
322 //cv::Mat_<qreal> I2;
323 //cv::GaussianBlur(I1, I2, cv::Size(3, 3), 1, 1);
324 //I1 = I2;
325
326 QVector<cv::Point> peaks;
327 for (int i = 0; i < crystalNum; ++i)
328 {
329 qreal minVal, maxVal;
330 cv::Point minLoc, maxLoc;
331 cv::minMaxLoc(I1, &minVal, &maxVal, &minLoc, &maxLoc);
332
333 peaks.append(maxLoc);
334
335 // 将maxLoc附近的元素置为0
336 int m0 = maxLoc.x;
337 int n0 = maxLoc.y;
338 int r = 6;
339
340 for (int n = n0-r; n <= n0+r; ++n)
341 {
342 for (int m = m0-r; m <= m0+r ; ++m)
343 {
344 if(cv::norm(maxLoc-cv::Point(m, n))<=r)
345 {
346 I1(n, m) = 0;
347 }
348 }
349 }
350 }
351
352 // sort peaks
353 for (int iRow = 0; iRow < nCrystal; ++iRow)
354 {
355 // 选出 y 值较小的 m_nCrystal 个点作为一行， y值升序排列
356 std::sort(peaks.begin(), peaks.end(),
357 [](const cv::Point& p1, const cv::Point& p2){return p1.y < p2.y;});
358
359 QVector<cv::Point> rowPeaks(peaks.begin(), peaks.begin()+nCrystal);
360 // 将他们按 x 值升序排列
361 std::sort(rowPeaks.begin(), rowPeaks.end(),
362 [](const cv::Point& p1, const cv::Point& p2){return p1.x < p2.x;});
363
364 for (int iCol = 0; iCol < nCrystal; ++iCol)
365 {
366 peaks.pop_front();
367 m_pt(iRow, iCol)[0] = rowPeaks[iCol].x;
368 m_pt(iRow, iCol)[1] = rowPeaks[iCol].y;
369 }
370 }
371}

395void Block::CalSegResult()
396{
397 //step 4, 开始分割, //计算像素点属于哪个峰
398 cv::Mat_<quint16> seg1(nPixel, nPixel); // 行信息
399 cv::Mat_<quint16> seg2(nPixel, nPixel); // 列信息
400
401 for (int i = 0; i < nPixel; ++i)
402 {
403 for (int j = 0; j < nPixel; ++j)
404 {
405 cv::Mat_<qreal> dis(nCrystal, nCrystal);
406
407 for (int iRow = 0; iRow < nCrystal; ++iRow)
408 {
409 for (int iCol = 0; iCol < nCrystal; ++iCol)
410 {
411 cv::Point2f p(m_pt(iRow, iCol)[0], m_pt(iRow, iCol)[1]);
412 dis(iRow, iCol) = cv::norm(cv::Point2f(j, i)-p);
413 }
414 }
415
416 cv::Point min_loc;
417 cv::minMaxLoc(dis, NULL, NULL, &min_loc, NULL);
418 seg1(i, j) = min_loc.y;
419 seg2(i, j) = min_loc.x;
420 }
421 }
422
423 m_segr = seg1*nCrystal + seg2;
424 m_edge = cv::Mat::zeros(nPixel, nPixel, CV_8UC1);
425 m_segMap = m_I0.clone();
426
427 qreal max_val;
428 cv::minMaxLoc(m_I0, NULL, &max_val, NULL, NULL);
429
430 for (int i = 1; i < nPixel-1; ++i)
431 {
432 for (int j = 1; j < nPixel-1; ++j)
433 {
434 int id0 = m_segr(i, j);
435 int id1 = m_segr(i+1, j);
436 int id2 = m_segr(i, j+1);
437 if(id0!=id1 || id0!=id2)
438 {
439 m_edge(i, j) = 1;
440 m_segMap(i, j) = max_val;
441 }
442 }
443 }
444
445 // 将每个BK 的分割结果保存成文件，以便后面调用
446 // 这样做的必要性是因为所有BK分割完才能生成整台机器的位置查找表
447 // 这样做还有一个优点，如果某个BK分割结果不理想，
448 // 后期还可以单独分割这个BK，然后生成位置查找表
449 QString fileName = "./Data/segResult.xml";
450
451 // 创建文件存储对象
452 cv::FileStorage fs(fileName.toStdString(), cv::FileStorage::WRITE);
453 fs.write("floodmap", m_I0); // 将 I0 写入
454 fs.write("edge", m_edge); // 写入边界结果
455 fs.write("segr", m_segr); // 写入分割结果
456 fs.release(); // 关闭文件
457}

483void Block::CalSegFOM()
484{
485 // step 5, 求分割质量参数
486 // calculate peaks mean-height
487
488 //cv::Mat I1;
489 //cv::blur(m_I0, I1, cv::Size(3, 3));
490
491 qreal peak_total_evts = 0;
492 qreal totalFWHM = 0;
493 for (int i = 0; i < nCrystal; ++i)
494 {
495 for (int j = 0; j < nCrystal; ++j)
496 {
497 int x = m_pt(i, j)[0];
498 int y = m_pt(i, j)[1];
499 peak_total_evts += m_I0(y, x);
500
501 // reduce 3*15 matrix to line, then find FWHM
502 // x direction
503 cv::Mat M1 = m_I0(cv::Rect(x-7, y-1, 15, 3));
504 qreal FWHM_x = CalFWHM(M1);
505
506 // y direction
507 cv::Mat M2 = m_I0(cv::Rect(x-1, y-7, 3, 15));
508 qreal FWHM_y = CalFWHM(M2.t());
509
510 qreal FWHM_xy = (FWHM_x + FWHM_y)/2;
511 totalFWHM += FWHM_xy;
512 }
513 }
514 qreal peak_mean_height = peak_total_evts/crystalNum;
515
516 // valley mean-height
517 quint64 edgeEvts = 0;
518 quint64 edgeN = 0;
519 for (int i = 0; i < nPixel; ++i)
520 {
521 for (int j = 0; j < nPixel; ++j)
522 {
523 if ( ymin<i && i<ymax && xmin<j && j<xmax )
524 {
525 int aN = static_cast<int>(m_I0(i, j));
526 if ( 1==m_edge(i, j) )
527 {
528 edgeN++;
529 edgeEvts += aN;
530 }
531 }
532 }
533 }
534
535 qreal valley_mean_height = 1.0*edgeEvts/edgeN;
536
537 m_PVR = peak_mean_height/valley_mean_height;
538 m_FWHM = totalFWHM/crystalNum;
539}

719qreal Block::CalFWHM(cv::Mat I)
720{
721 cv::Mat col_sum;
722 cv::reduce(I, col_sum, 0, cv::REDUCE_SUM);
723
724 cv::Mat resizedMat;
725 cv::resize(col_sum, resizedMat,
726 cv::Size(I.cols*10, 1), 0, 0, cv::INTER_LINEAR);
727
728 cv::Mat blurredMat;
729 cv::blur(resizedMat, blurredMat, cv::Size(10, 1));
730
731 QVector<qreal> v = {blurredMat.begin<qreal>(),
732 blurredMat.end<qreal>()};
733
734 int peakIdx = std::max_element(v.begin(), v.end()) - v.begin();
735 qreal peakValue = v[peakIdx];
736
737 qreal halfMax = peakValue/2;
738 int leftIdx = peakIdx;
739 int rightIdx = peakIdx;
740
741 while (leftIdx>0 && v[leftIdx]>halfMax)
742 {
743 leftIdx--;
744 }
745
746 while (rightIdx<v.size()-1 && v[rightIdx]>halfMax)
747 {
748 rightIdx++;
749 }
750
751 qreal FWHM = (rightIdx - leftIdx)/10.0; // 半高宽
752 return FWHM;
753}

783
784 // 在 2 到 √n 之间搜索因子

786 {
787 if (nCrystal % i == 0)

792

794 if (diff < minDiff)

796 minDiff = diff;
797 n1 = factor1;
798 n2 = factor2;
799 }
800 }
801 }
